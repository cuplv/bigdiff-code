CREATE KEYSPACE bigdiff WITH replication = {
  'class': 'SimpleStrategy',
  'replication_factor': '1'
};

USE bigdiff;

create table repo_info(id int, name text, full_name text, owner_login text, owner_id int, owner_url text, private boolean, html_url text, description text, url text, created_at timestamp, updated_at timestamp, pushed_at timestamp, clone_url text, size int, stargazers_count int, watchers_count int, language text, has_issues boolean, forks_count int, open_issues_count int, score float, primary key(id));
